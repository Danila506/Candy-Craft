
model User {
  id    String @id @default(uuid())
  name  String
  email String @unique
  cart  Cart?

  @@map("user")
}

model Product {
  id          Int    @id @default(autoincrement())
  name        String
  description String
  price       Int
  inStock     Int    @default(0) @map("in_stock")
  imageUrl    String

  categoryId Int
  category   Category @relation(fields: [categoryId], references: [id])

  createdAt DateTime   @default(now())
  updatedAt DateTime   @updatedAt
  cartItems CartItem[]
  cartId    Int?

  @@index([categoryId])
  @@map("products")
}

model Category {
  id          Int    @id @default(autoincrement())
  name        String @unique
  description String
  imageUrl    String

  product Product[]

  @@map("category")
}

model Cart {
  id     String @id @default(uuid())
  userId String @unique
  user   User   @relation(fields: [userId], references: [id])

  // Связь One-to-Many с CartItem
  items CartItem[]

  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt
}

// Промежуточная модель для хранения количества
model CartItem {
  id     String @id @default(uuid())
  cartId String
  cart   Cart   @relation(fields: [cartId], references: [id])

  productId Int
  product   Product @relation(fields: [productId], references: [id])

  quantity Int @default(1) // Количество товара

  @@unique([cartId, productId]) // Один товар в корзине один раз
  @@index([cartId])
  @@index([productId])
}
